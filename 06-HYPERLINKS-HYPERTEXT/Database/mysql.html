What is MySQL?

MySQL is an open-source relational database management system (RDBMS) based on Structured Query Language (SQL). It's one of the most popular databases in the world, widely used for web applications and online publishing. MySQL is known for its reliability, performance, and ease of use.

üéØ Key Features of MySQL

  • Open Source: Free to use and modify
  • Cross-Platform: Runs on multiple operating systems
  • ACID Compliant: Ensures data integrity and consistency
  • High Performance: Fast query processing and optimization
  • Scalable: Handles large amounts of data efficiently
  • Secure: Built-in security features and user management
  • Reliable: Proven stability in production environments

üìä Database Design and Tables

Creating Tables:

-- Create users table
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    age INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Create posts table
CREATE TABLE posts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    user_id INT,
    published BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Create categories table
CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- Create many-to-many relationship table
CREATE TABLE post_categories (
    post_id INT,
    category_id INT,
    PRIMARY KEY (post_id, category_id),
    FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
    FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE CASCADE
);

Modifying Tables:

-- Add column
ALTER TABLE users ADD COLUMN phone VARCHAR(20);

-- Modify column
ALTER TABLE users MODIFY COLUMN name VARCHAR(150);

-- Drop column
ALTER TABLE users DROP COLUMN phone;

-- Add index
CREATE INDEX idx_email ON users(email);

-- Add foreign key
ALTER TABLE posts ADD CONSTRAINT fk_user 
FOREIGN KEY (user_id) REFERENCES users(id);

üîç SQL Queries (CRUD Operations)

INSERT - Creating Data:

-- Insert single record
INSERT INTO users (name, email, password, age) 
VALUES ('John Doe', 'john@example.com', 'hashed_password', 30);

-- Insert multiple records
INSERT INTO users (name, email, password, age) VALUES
('Jane Smith', 'jane@example.com', 'hashed_password', 25),
('Bob Johnson', 'bob@example.com', 'hashed_password', 35),
('Alice Brown', 'alice@example.com', 'hashed_password', 28);

-- Insert with SELECT
INSERT INTO categories (name, description)
SELECT DISTINCT category, 'Auto-generated category' 
FROM temp_posts;

SELECT - Reading Data:

-- Basic SELECT
SELECT * FROM users;
SELECT name, email FROM users;

-- WHERE clause
SELECT * FROM users WHERE age > 25;
SELECT * FROM users WHERE name LIKE 'J%';
SELECT * FROM users WHERE age BETWEEN 25 AND 35;

-- ORDER BY
SELECT * FROM users ORDER BY name ASC;
SELECT * FROM users ORDER BY created_at DESC;

-- LIMIT
SELECT * FROM users LIMIT 10;
SELECT * FROM users LIMIT 5 OFFSET 10;

-- GROUP BY and HAVING
SELECT age, COUNT(*) as count 
FROM users 
GROUP BY age 
HAVING count > 1;

UPDATE - Modifying Data:

-- Update single record
UPDATE users 
SET age = 31 
WHERE id = 1;

-- Update multiple records
UPDATE users 
SET age = age + 1 
WHERE created_at < '2023-01-01';

-- Update with JOIN
UPDATE users u
JOIN posts p ON u.id = p.user_id
SET u.name = CONCAT(u.name, ' (Author)')
WHERE p.published = TRUE;

DELETE - Removing Data:

-- Delete specific records
DELETE FROM users WHERE age < 18;

-- Delete with JOIN
DELETE u FROM users u
LEFT JOIN posts p ON u.id = p.user_id
WHERE p.id IS NULL;

-- Truncate table (faster for all records)
TRUNCATE TABLE temp_data;

üîó Advanced SQL Concepts

JOINs:

-- INNER JOIN
SELECT u.name, p.title
FROM users u
INNER JOIN posts p ON u.id = p.user_id;

-- LEFT JOIN
SELECT u.name, COUNT(p.id) as post_count
FROM users u
LEFT JOIN posts p ON u.id = p.user_id
GROUP BY u.id, u.name;

-- RIGHT JOIN
SELECT u.name, p.title
FROM users u
RIGHT JOIN posts p ON u.id = p.user_id;

-- Multiple JOINs
SELECT u.name, p.title, c.name as category
FROM users u
JOIN posts p ON u.id = p.user_id
JOIN post_categories pc ON p.id = pc.post_id
JOIN categories c ON pc.category_id = c.id;

Subqueries:

-- Subquery in WHERE
SELECT * FROM users 
WHERE id IN (SELECT user_id FROM posts WHERE published = TRUE);

-- Subquery in SELECT
SELECT name, 
       (SELECT COUNT(*) FROM posts WHERE user_id = users.id) as post_count
FROM users;

-- EXISTS
SELECT * FROM users u
WHERE EXISTS (SELECT 1 FROM posts p WHERE p.user_id = u.id);

Window Functions (MySQL 8.0+):

-- ROW_NUMBER
SELECT name, age,
       ROW_NUMBER() OVER (ORDER BY age DESC) as rank
FROM users;

-- RANK and DENSE_RANK
SELECT name, age,
       RANK() OVER (ORDER BY age DESC) as rank,
       DENSE_RANK() OVER (ORDER BY age DESC) as dense_rank
FROM users;

-- Partition by
SELECT name, age, department,
       AVG(age) OVER (PARTITION BY department) as dept_avg_age
FROM employees;

üîê Security and User Management

User Management:

-- Create user
CREATE USER 'appuser'@'localhost' IDENTIFIED BY 'strong_password';

-- Grant privileges
GRANT SELECT, INSERT, UPDATE, DELETE ON myapp.* TO 'appuser'@'localhost';

-- Grant all privileges
GRANT ALL PRIVILEGES ON myapp.* TO 'admin'@'localhost';

-- Show grants
SHOW GRANTS FOR 'appuser'@'localhost';

-- Revoke privileges
REVOKE DELETE ON myapp.* FROM 'appuser'@'localhost';

-- Drop user
DROP USER 'appuser'@'localhost';

-- Flush privileges
FLUSH PRIVILEGES;

Security Best Practices:

  • Use strong passwords: Complex passwords for database users
  • Principle of least privilege: Grant minimum required permissions
  • Regular updates: Keep MySQL updated
  • Secure connections: Use SSL/TLS for connections
  • Backup regularly: Regular database backups
  • Monitor access: Log and monitor database access

üíæ Backup and Recovery

Backup Commands:

-- Backup single database
mysqldump -u username -p myapp > myapp_backup.sql

-- Backup multiple databases
mysqldump -u username -p --databases db1 db2 > multiple_backup.sql

-- Backup all databases
mysqldump -u username -p --all-databases > all_backup.sql

-- Backup with compression
mysqldump -u username -p myapp | gzip > myapp_backup.sql.gz

-- Backup specific tables
mysqldump -u username -p myapp users posts > tables_backup.sql

Restore Commands:

-- Restore database
mysql -u username -p myapp < myapp_backup.sql

-- Restore compressed backup
gunzip < myapp_backup.sql.gz | mysql -u username -p myapp

-- Create database and restore
mysql -u username -p -e "CREATE DATABASE myapp_restored;"
mysql -u username -p myapp_restored < myapp_backup.sql

üåê MySQL with Web Applications

Node.js Integration:

const mysql = require('mysql2/promise');

// Create connection
const connection = await mysql.createConnection({
    host: 'localhost',
    user: 'appuser',
    password: 'password',
    database: 'myapp'
});

// Execute queries
const [rows] = await connection.execute(
    'SELECT * FROM users WHERE age > ?',
    [25]
);

// Insert data
const [result] = await connection.execute(
    'INSERT INTO users (name, email, age) VALUES (?, ?, ?)',
    ['John Doe', 'john@example.com', 30]
);

console.log('Inserted ID:', result.insertId);

Python Integration:

import mysql.connector

# Create connection
connection = mysql.connector.connect(
    host='localhost',
    user='appuser',
    password='password',
    database='myapp'
)

cursor = connection.cursor()

# Execute query
cursor.execute("SELECT * FROM users WHERE age > %s", (25,))
results = cursor.fetchall()

for row in results:
    print(row)

# Insert data
cursor.execute(
    "INSERT INTO users (name, email, age) VALUES (%s, %s, %s)",
    ('Jane Doe', 'jane@example.com', 28)
)

connection.commit()
connection.close()

üîß MySQL Best Practices

  1. Use appropriate data types: Choose the right data type for each column
  2. Normalize your database: Reduce data redundancy
  3. Use indexes wisely: Index frequently queried columns
  4. Avoid SELECT *: Select only needed columns
  5. Use prepared statements: Prevent SQL injection
  6. Regular maintenance: Optimize and analyze tables regularly
  7. Monitor performance: Use slow query log
  8. Backup regularly: Implement backup strategy

‚ö° MySQL Performance Tuning

Configuration Optimization:

-- Key MySQL configuration variables
SET GLOBAL innodb_buffer_pool_size = 1073741824; -- 1GB
SET GLOBAL query_cache_size = 67108864; -- 64MB
SET GLOBAL max_connections = 200;
SET GLOBAL innodb_log_file_size = 268435456; -- 256MB

-- Show current configuration
SHOW VARIABLES LIKE 'innodb_buffer_pool_size';
SHOW VARIABLES LIKE 'query_cache%';

Query Optimization Tips:

  • Use LIMIT: Limit result sets when possible
  • Avoid functions in WHERE: Don't use functions on indexed columns
  • Use EXISTS instead of IN: For better performance with subqueries
  • Optimize JOINs: Ensure proper indexes on JOIN columns
  • Use UNION ALL: Instead of UNION when duplicates are acceptable
